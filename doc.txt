3.3 main.py
from google.cloud import bigquery

def run_query():
    client = bigquery.Client()

    query = """
    SELECT title, score, view_count
    FROM `bigquery-public-data.stackoverflow.posts_questions`
    LIMIT 5
    """

    results = client.query(query)

    for row in results:
        print(row.title, row.score, row.view_count)

if __name__ == "__main__":
    run_query()

STEP 5 — SQL-Based Tests (0 rows = PASS)
Example SQL Test

queries/test_no_negative_views.sql

SELECT id
FROM `bigquery-public-data.stackoverflow.posts_questions`
WHERE view_count < 0
Python Test

tests/test_no_negative_views.py

from src.bigquery_client import get_client
from src.query_runner import load_query

def test_no_negative_views():
    client = get_client()
    query = load_query("queries/test_no_negative_views.sql")
    results = list(client.query(query))
    assert len(results) == 0
